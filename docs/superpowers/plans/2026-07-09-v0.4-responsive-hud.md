# v0.4 Responsive HUD Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the mixed fixed-coordinate and anchored HUD with one responsive Control layout that keeps status bars, quick slots, and menu controls aligned at 1280 x 720, 1600 x 900, and the Godot embedded preview size.

**Architecture:** Keep `HUD` as a `CanvasLayer`, but place every visible element under one full-viewport `Control` root. A bottom-anchored panel owns a background `TextureRect` and an `HBoxContainer` with fixed left and right regions plus a flexible center region; `hud.gd` updates values only and does not calculate screen coordinates.

**Tech Stack:** Godot 4.x, GDScript, Control containers, TextureRect, ProgressBar, headless SceneTree tests, GUI screenshot verification.

- [ ] **Step 1: Create the responsive layout test**

Create a SceneTree test that instantiates a fresh HUD inside a `SubViewport` for each required size. The test must use the new hierarchy expected from the design so that it fails against the current scene:

```gdscript
extends SceneTree

const HudScene := preload("res://scenes/ui/hud.tscn")
const TEST_SIZES := [
	Vector2i(1280, 720),
	Vector2i(1600, 900),
	Vector2i(1498, 843),
]


func _initialize() -> void:
	var failures: Array[String] = []
	for test_size in TEST_SIZES:
		await _test_layout_at_size(test_size, failures)

	if failures.is_empty():
		print("hud_responsive_layout_test: PASS")
		quit(0)
		return

	for failure in failures:
		push_error(failure)
	quit(1)


func _test_layout_at_size(test_size: Vector2i, failures: Array[String]) -> void:
	var viewport := SubViewport.new()
	viewport.size = test_size
	root.add_child(viewport)

	var hud := HudScene.instantiate()
	viewport.add_child(hud)
	await process_frame

	var bottom_panel := hud.get_node("LayoutRoot/BottomPanel") as Control
	var left_region := hud.get_node("LayoutRoot/BottomPanel/ContentMargin/Regions/LeftRegion") as Control
	var center_region := hud.get_node("LayoutRoot/BottomPanel/ContentMargin/Regions/CenterRegion") as Control
	var right_region := hud.get_node("LayoutRoot/BottomPanel/ContentMargin/Regions/RightRegion") as Control
	var quick_slots := hud.get_node("LayoutRoot/BottomPanel/ContentMargin/Regions/CenterRegion/QuickSlots") as HBoxContainer
	var menu_buttons := hud.get_node("LayoutRoot/BottomPanel/ContentMargin/Regions/RightRegion/MenuButtons") as HBoxContainer
	var status_label := hud.get_node("LayoutRoot/BottomPanel/ContentMargin/Regions/RightRegion/StatusLabel") as Label

	_expect(bottom_panel.get_global_rect().end.y <= test_size.y + 0.5, "%s bottom panel must stay in viewport" % test_size, failures)
	_expect(left_region.get_global_rect().end.x <= center_region.get_global_rect().position.x + 0.5, "%s left and center regions must not overlap" % test_size, failures)
	_expect(center_region.get_global_rect().end.x <= right_region.get_global_rect().position.x + 0.5, "%s center and right regions must not overlap" % test_size, failures)
	_expect(right_region.get_global_rect().end.x <= test_size.x + 0.5, "%s right region must stay in viewport" % test_size, failures)
	_expect(quick_slots.get_child_count() == 6, "%s quick bar must keep six slots" % test_size, failures)
	_expect(not menu_buttons.get_global_rect().intersects(status_label.get_global_rect()), "%s menu and status must not overlap" % test_size, failures)

	viewport.queue_free()
	await process_frame


func _expect(condition: bool, message: String, failures: Array[String]) -> void:
	if not condition:
		failures.append(message)
```

- [ ] **Step 1: Replace the fixed visual root**

In `hud.tscn`, remove `GeneratedUiFrame` and make the scene follow this hierarchy:

```text
HUD (CanvasLayer)
└── LayoutRoot (Control, full rect)
    ├── BottomPanel (Control, bottom wide, 145 px high)
    │   ├── Background (TextureRect, full rect)
    │   └── ContentMargin (MarginContainer, full rect)
    │       └── Regions (HBoxContainer)
    │           ├── LeftRegion (PanelContainer, 280 px minimum)
    │           │   └── StatusBars (VBoxContainer)
    │           │       ├── HpRow (HBoxContainer)
    │           │       │   ├── HpLabel
    │           │       │   ├── HpBar (ProgressBar)
    │           │       │   └── HpValue
    │           │       └── MpRow (HBoxContainer)
    │           │           ├── MpLabel
    │           │           ├── MpBar (ProgressBar)
    │           │           └── MpValue
    │           ├── CenterRegion (CenterContainer, horizontal expand)
    │           │   └── QuickSlots (HBoxContainer)
    │           │       └── Slot1 ... Slot6
    │           └── RightRegion (VBoxContainer, 420 px minimum)
    │               ├── StatusLabel
    │               └── MenuButtons (HBoxContainer)
    │                   ├── InventoryButton
    │                   └── EquipmentButton
    └── CharacterPortrait (TextureRect, hidden)
```

Use these critical layout properties:

```ini
[node name="LayoutRoot" type="Control" parent="."]
anchors_preset = 15
anchor_right = 1.0
anchor_bottom = 1.0
grow_horizontal = 2
grow_vertical = 2

[node name="BottomPanel" type="Control" parent="LayoutRoot"]
anchors_preset = 12
anchor_top = 1.0
anchor_right = 1.0
anchor_bottom = 1.0
offset_top = -145.0
grow_horizontal = 2
grow_vertical = 0

[node name="Background" type="TextureRect" parent="LayoutRoot/BottomPanel"]
anchors_preset = 15
anchor_right = 1.0
anchor_bottom = 1.0
grow_horizontal = 2
grow_vertical = 2
texture = ExtResource("1_ui_atlas")
expand_mode = 1
stretch_mode = 0

[node name="LeftRegion" type="PanelContainer" parent="LayoutRoot/BottomPanel/ContentMargin/Regions"]
custom_minimum_size = Vector2(280, 0)
layout_mode = 2

[node name="CenterRegion" type="CenterContainer" parent="LayoutRoot/BottomPanel/ContentMargin/Regions"]
layout_mode = 2
size_flags_horizontal = 3

[node name="RightRegion" type="VBoxContainer" parent="LayoutRoot/BottomPanel/ContentMargin/Regions"]
custom_minimum_size = Vector2(420, 0)
layout_mode = 2
```

Configure each `ProgressBar` with `show_percentage = false`, stable minimum width, and values in the 0 to 100 range. Keep the six quick slots at 40 x 40 so content changes cannot resize the bar.

- [ ] **Step 3: Update `hud.gd` node types and paths**

Replace `ColorRect` bar fields with `ProgressBar` and resolve nodes under the new hierarchy:

```gdscript
const CONTENT_PATH := "LayoutRoot/BottomPanel/ContentMargin/Regions"

var hp_value: Label
var mp_value: Label
var hp_bar: ProgressBar
var mp_bar: ProgressBar
var status_label: Label
var quick_slots: HBoxContainer


func set_health(current: int, maximum: int) -> void:
	_ensure_nodes()
	if hp_value == null or hp_bar == null:
		return
	hp_value.text = "%d/%d" % [current, maximum]
	hp_bar.max_value = max(maximum, 1)
	hp_bar.value = clampi(current, 0, max(maximum, 0))


func set_mana(current: int, maximum: int) -> void:
	_ensure_nodes()
	if mp_value == null or mp_bar == null:
		return
	mp_value.text = "%d/%d" % [current, maximum]
	mp_bar.max_value = max(maximum, 1)
	mp_bar.value = clampi(current, 0, max(maximum, 0))


func _ensure_nodes() -> void:
	if hp_value == null:
		hp_value = get_node_or_null(CONTENT_PATH + "/LeftRegion/StatusBars/HpRow/HpValue") as Label
	if mp_value == null:
		mp_value = get_node_or_null(CONTENT_PATH + "/LeftRegion/StatusBars/MpRow/MpValue") as Label
	if hp_bar == null:
		hp_bar = get_node_or_null(CONTENT_PATH + "/LeftRegion/StatusBars/HpRow/HpBar") as ProgressBar
	if mp_bar == null:
		mp_bar = get_node_or_null(CONTENT_PATH + "/LeftRegion/StatusBars/MpRow/MpBar") as ProgressBar
	if status_label == null:
		status_label = get_node_or_null(CONTENT_PATH + "/RightRegion/StatusLabel") as Label
	if quick_slots == null:
		quick_slots = get_node_or_null(CONTENT_PATH + "/CenterRegion/QuickSlots") as HBoxContainer
```

Keep `set_experience()`, `set_gold()`, `set_level()`, and the public method names unchanged.
